rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    
    // User profiles
    match /users/{userId} {
      allow read: if request.auth != null;
      allow write: if request.auth != null && request.auth.uid == userId;
    }

    // Chats
    match /chats/{chatId} {
      allow read: if request.auth != null && request.auth.uid in resource.data.participants;
      allow update: if request.auth != null && request.auth.uid in resource.data.participants
                    && (
                      (!request.resource.data.diff(resource.data).affectedKeys().hasAny(['participants', 'admins'])) 
                      || (request.auth.uid in resource.data.get('admins', []))
                    );
      allow create: if request.auth != null && request.auth.uid in request.resource.data.participants;
      
      // Messages within a chat
      match /messages/{messageId} {
        allow read: if request.auth != null && request.auth.uid in get(/databases/$(database)/documents/chats/$(chatId)).data.participants;
        allow create: if request.auth != null 
                     && request.auth.uid in get(/databases/$(database)/documents/chats/$(chatId)).data.participants
                     && request.resource.data.senderId == request.auth.uid;
                     
        // Only allow sender to update/delete their own message, or users in chat to update readBy/starredBy etc.
        allow update: if request.auth != null && request.auth.uid in get(/databases/$(database)/documents/chats/$(chatId)).data.participants;
        allow delete: if request.auth != null && request.resource.data.senderId == request.auth.uid;
      }
    }
    
    // WebRTC Calls Signaling
    match /calls/{callId} {
      allow read: if request.auth != null;
      allow write: if request.auth != null;
      
      match /callerCandidates/{id} {
        allow read, write: if request.auth != null;
      }
      match /calleeCandidates/{id} {
        allow read, write: if request.auth != null;
      }
    }

    // Call History
    match /call_history/{callId} {
      allow read: if request.auth != null && request.auth.uid in resource.data.participants;
      allow create: if request.auth != null && request.auth.uid in request.resource.data.participants;
    }
  }
}
